(() => {

    item.querySelector('.ww-faq-question').setAttribute('aria-expanded','true');
  };
  const closeItem = item => {
    item.classList.remove('is-open');
    item.querySelector('.ww-faq-question').setAttribute('aria-expanded','false');
  };
  items.forEach(item => item.querySelector('.ww-faq-question').addEventListener('click', () => item.classList.contains('is-open') ? closeItem(item) : openItem(item)));
  root.querySelectorAll('[data-faq-target]').forEach(button => button.addEventListener('click', () => {
    const item = root.querySelector('#' + CSS.escape(button.dataset.faqTarget));
    if (!item) return;
    openItem(item);
    item.scrollIntoView({behavior:'smooth',block:'center'});
  }));
  root.querySelectorAll('[data-category-target]').forEach(button => button.addEventListener('click', () => {
    const section = root.querySelector('#' + CSS.escape(button.dataset.categoryTarget));
    section?.scrollIntoView({behavior:'smooth',block:'start'});
  }));
  const normalize = value => value.toLowerCase().replace(/\s+/g,' ').trim();
  const filter = () => {
    const query = normalize(input.value);
    const terms = query.split(' ').filter(Boolean);
    let visible = 0;
    items.forEach(item => {
      const haystack = normalize(item.textContent + ' ' + (item.dataset.search || ''));
      const match = !terms.length || terms.every(term => haystack.includes(term));
      item.hidden = !match;
      if (match) visible++;
      if (terms.length && match) openItem(item); else if (!terms.length) closeItem(item);
    });
    categories.forEach(category => category.hidden = ![...category.querySelectorAll('[data-faq-item]')].some(item => !item.hidden));
    popular.hidden = terms.length > 0;
    nav.hidden = terms.length > 0;
    empty.classList.toggle('is-visible', visible === 0);
    count.textContent = terms.length ? `${visible} matching question${visible === 1 ? '' : 's'}` : `Showing all ${items.length} questions`;
  };
